import java.util.*;

class Solution {
    public int solution(int[] stones, int k) {
        int hi = Integer.MIN_VALUE;
        int lo = Integer.MAX_VALUE;
        
        for (int stone : stones) {
            hi = Math.max(hi, stone);
            lo = Math.min(lo, stone);
        }
        
        if (hi == lo) return lo;
        
        while (lo < hi) {
            int mid = lo + (hi - lo) / 2;
            
            if (canCross(stones, k, mid)) {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        
        return lo - 1;
    }
    
    private boolean canCross(int[] stones, int k, int friends) {
        int passCount = 0;
        
        for (int stone : stones) {
            if (stone - friends < 0) {
                passCount++;
            } else {
                passCount = 0;
            }
            
            if (passCount == k) return false;
        }
        
        return true;
    }
}
